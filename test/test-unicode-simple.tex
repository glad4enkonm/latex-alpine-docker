\section{Basic Unicode Support}
This document tests Unicode characters that should work with default fonts.

\subsection{European Characters}
\begin{itemize}
    \item French: cafe, naive, resume
    \item German: Grosse, weiss, Strasse
    \item Spanish: nino, ano, corazon
    \item Nordic: Asa, Kobenhavn, Malmo
\end{itemize}

\subsection{Mathematical Unicode}
\begin{itemize}
    \item Basic: +- x / <= >= != ~ infinity
    \item Arrows: -> <- up down => <= <=>
\end{itemize}

\subsection{Simple Symbols}
\begin{itemize}
    \item Currency: EUR GBP YEN
    \item Legal: (C) (R) TM
    \item Temperature: degrees C, degrees F
\end{itemize}

\section{Compilation}

\begin{verbatim}

\end{verbatim}
or:
\begin{verbatim}
lualatex test-unicode-simple.tex
\end{verbatim}
